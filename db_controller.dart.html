// Method to authenticate user credentials
Future<bool> authenticate(String email, String pass) async {
  final conn = await MySqlConnection.connect(settings);
  try {
    // Query to check if the user exists with the provided email and password
    var result = await conn.query(
        'SELECT name FROM users WHERE email = ? AND pass = ?',
        [email, pass]
    );

    // If result has any rows, the credentials are valid
    if (result.isNotEmpty) {
      // Extract the user's name and convert it to a String
      globalUserName = result.first['name'].toString(); // Ensure it's a string
      return true;
    } else {
      return false;
    }
  } catch (e) {
    print('Error during authentication: $e');
    return false; // Return false in case of error
  } finally {
    await conn.close(); // Ensure the connection is closed
  }
}

Future<List<Map<String, dynamic>>> fetchMenuItems(int shopId) async {
  List<Map<String, dynamic>> menuItems = [];
  var conn = await MySqlConnection.connect(settings);

  try {
    var results = await conn.query(
      'SELECT menu_name, price FROM menu WHERE shop_id = ?',
      [shopId],
    );

    for (var row in results) {
      // Ensure that the menu_name is converted to a String if it comes as a BLOB
      String menuName = (row[0] is List<int>)
          ? String.fromCharCodes(row[0]) // Convert BLOB to String
          : row[0].toString(); // Ensure it is a String

      // Check if the price is of a valid type (int or double)
      double price = row[1] is double ? row[1] : double.tryParse(row[1].toString()) ?? 0.0;

      menuItems.add({
        'menu_name': menuName,
        'price': price,
      });
    }
  } catch (e) {
    print('Error fetching menu items: $e');
  } finally {
    await conn.close();
  }
  return menuItems;
}

// Function to insert order records into the database
Future<void> insertOrderRecords(String username, List<Map<String, dynamic>> cartItems, String totalPrice, String fee) async {
  // Establish the connection
  final conn = await MySqlConnection.connect(settings);

  try {
    for (var item in cartItems) {
      // Define the SQL query to insert an order record
      final String query = '''
        INSERT INTO orders (name, menu_name, quantity, shop_id, total_price, fee) 
        VALUES (?, ?, ?, ?, ?, ?) 
      ''';

      // Execute the query with values for each item in the cart
      await conn.query(
        query,
        [username, item['menu_name'], item['quantity'], item['shop_id'], totalPrice, fee], // Get shop_id from the cart item
      );
    }
    print('Order records inserted successfully.');
  } catch (e) {
    print('Error inserting order records: $e');
  } finally {
    await conn.close(); // Close the connection
  }
}

Future<void> dbAddMenuItem(String menuName, double price, int shopId, int sellerId) async {
  final conn = await MySqlConnection.connect(settings);
  try {
    // First, fetch the shop_name based on shop_id from the sellers table
    var result = await conn.query(
      'SELECT shop_name FROM sellers WHERE shop_id = ?',
      [shopId],
    );

    // Check if shop_name is found
    if (result.isNotEmpty) {
      String shopName = result.first[0].toString();

      // Insert new menu item into menu table with the fetched shop_name
      await conn.query(
        'INSERT INTO menu (menu_name, price, shop_id, seller_id, shop_name) VALUES (?, ?, ?, ?, ?)',
        [menuName, price, shopId, sellerId, shopName],
      );

      print('Menu item added: $menuName, Price: $price, Shop Name: $shopName');
    } else {
      print('No shop name found for the provided shop_id.');
    }
  } catch (e) {
    print('Error adding menu item: $e');
  } finally {
    await conn.close();
  }
}

// Function to fetch shop info (shop_name and payment_address)
Future<Map<String, dynamic>?> dbFetchShopInfo(int shopId) async {
  final conn = await MySqlConnection.connect(settings);
  try {
    var results = await conn.query(
        'SELECT shop_name, payment_address FROM sellers WHERE shop_id = ?', [shopId]
    );

    if (results.isNotEmpty) {
      var row = results.first;
      return {
        'shop_name': row[0] is Blob ? String.fromCharCodes(row[0].toBytes()) : row[0], // Convert Blob to String if needed
        'payment_address': row[1] is Blob ? String.fromCharCodes(row[1].toBytes()) : row[1], // Convert Blob to String if needed
      };
    }
  } catch (e) {
    print('Error fetching shop info: $e');
  } finally {
    await conn.close();
  }
  return null;
}

// Fetch orders for a specific shop
Future<List<Map<String, dynamic>>> fetchOrdersForShop(int shopId) async {
  final conn = await MySqlConnection.connect(settings);
  try {
    var results = await conn.query(
        'SELECT name, menu_name, quantity '
            'FROM orders '
            'WHERE shop_id = ?',
        [shopId]
    );

    // Transform the results into a list of maps
    return results.map((row) {
      return {
        'name': row[0],
        'menu_name': row[1],
        'quantity': row[2],
      };
    }).toList();
  } catch (e) {
    print('Error fetching orders: $e');
    return [];
  } finally {
    await conn.close(); // Ensure the connection is closed
  }
}
