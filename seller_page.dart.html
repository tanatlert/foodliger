import 'package:flutter/material.dart';
import 'package:foodliger/components/bottom_shop_nav_bar_seller.dart';
import 'package:foodliger/pages/shop_list_page.dart';
import 'package:foodliger/pages/cart_page.dart';
import 'package:foodliger/pages/seller/menu_list_page.dart';
import 'package:foodliger/pages/seller/order_page.dart';
import 'package:foodliger/pages/order_checker.dart'; // Import the OrderChecker class
import 'package:foodliger/database/db_controller.dart';

class SellerPage extends StatefulWidget {
  const SellerPage({super.key});

  @override
  State<SellerPage> createState() => _ShopPageState();
}

class _ShopPageState extends State<SellerPage> {
  // Variable
  int _selectedIndex = 0;
  OrderChecker? orderChecker; // Declare the OrderChecker

  // This method will update our selected index
  // when the user taps on the signin option
  void navigateShop(int index) {
    setState(() {
      _selectedIndex = index;
    });
  }

  // Pages to display
  final List<Widget> _pages = [
    SellerMenuControlPage(),
    OrderPage(),
  ];

  @override
  void initState() {
    super.initState();
    orderChecker = OrderChecker(); // Instantiate OrderChecker
    orderChecker!.startCheckingForOrders(globalSellerId!); // Start checking for new orders
  }

  @override
  void dispose() {
    orderChecker?.stopChecking(); // Stop checking for new orders when disposing
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: Colors.orange[300],
      appBar: AppBar(
        backgroundColor: Colors.orange[300],
        title: Text("Food Liger"),
        centerTitle: true,
        leading: IconButton(
          icon: Icon(Icons.logout),
          onPressed: () {
            Navigator.pushNamed(context, '/');
          },
        ),
      ),
      bottomNavigationBar: ShopOptionSeller(
        onTabChange: (index) => navigateShop(index),
      ),
      body: _pages[_selectedIndex],
    );
  }
}
